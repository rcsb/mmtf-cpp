153 BinaryDecoder::BinaryDecoder(const msgpack::object& obj, const std::string& key)
154  : key_(key) {
155  // sanity checks
156  if (obj.type != msgpack::type::BIN) {
157  throw DecodeError("The '" + key + "' entry is not binary data");
158  }
159  if (obj.via.bin.size < 12) {
160  std::stringstream err;
161  err << "The '" + key + "' entry is too short " << obj.via.bin.size;
162  throw DecodeError(err.str());
163  }
164  // get data (encoded data is only pointed to and not parsed here)
165  const char* bytes = obj.via.bin.ptr;
166  assignBigendian4(&strategy_, bytes);
167  assignBigendian4(&length_, bytes + 4);
168  assignBigendian4(&parameter_, bytes + 8);
169  encodedData_ = bytes + 12;
170  encodedDataLength_ = obj.via.bin.size - 12;
171 }

178 template<>
179 void BinaryDecoder::decode(std::vector<float>& output) {
180  // check strategy to parse
181  switch (strategy_) {
182  case 1: {
183  decodeFromBytes_(output);
184  break;
185  }
186  case 9: {
187  std::vector<int32_t> step1;
188  std::vector<int32_t> step2;
189  decodeFromBytes_(step1);
190  runLengthDecode_(step1, step2);
191  decodeDivide_(step2, parameter_, output);
192  break;
193  }
194  case 10: {
195  std::vector<int16_t> step1;
196  std::vector<int32_t> step2;
197  decodeFromBytes_(step1);
198  recursiveIndexDecode_(step1, step2);
199  deltaDecode_(step2);
200  decodeDivide_(step2, parameter_, output);
201  break;
202  }
203  case 11: {
204  std::vector<int16_t> step1;
205  decodeFromBytes_(step1);
206  decodeDivide_(step1, parameter_, output);
207  break;
208  }
209  case 12: {
210  std::vector<int16_t> step1;
211  std::vector<int32_t> step2;
212  decodeFromBytes_(step1);
213  recursiveIndexDecode_(step1, step2);
214  decodeDivide_(step2, parameter_, output);
215  break;
216  }
217  case 13: {
218  std::vector<int8_t> step1;
219  std::vector<int32_t> step2;
220  decodeFromBytes_(step1);
221  recursiveIndexDecode_(step1, step2);
222  decodeDivide_(step2, parameter_, output);
223  break;
224  }
225  default: {
226  std::stringstream err;
227  err << "Invalid strategy " << strategy_ << " for binary '" + key_
228  << "': does not decode to float array";
229  throw DecodeError(err.str());
230  }
231  }
232 
233  // check size
234  checkLength_(output.size());
235 }

424 // run length decoding
425 template<typename Int, typename IntOut>
426 void BinaryDecoder::runLengthDecode_(const std::vector<Int>& input,
427  std::vector<IntOut>& output) {
428  // we work with pairs of numbers
429  checkDivisibleBy_(2);
430  // find out size of resulting vector (for speed)
431  size_t out_size = 0;
432  for (size_t i = 0; i < input.size(); i += 2) {
433  out_size += input[i + 1];
434  }
435  // reserve space (for speed)
436  output.clear();
437  output.reserve(out_size);
438  // get data
439  for (size_t i = 0; i < input.size(); i += 2) {
440  const IntOut value = IntOut(input[i]);
441  const Int number = input[i+1];
442  for (Int j = 0; j < number; ++j) {
443  output.push_back(value);
444  }
445  }
446 }

449 template<typename Int>
450 void BinaryDecoder::deltaDecode_(const std::vector<Int>& input,
451  std::vector<Int>& output) {
452  // reserve space (for speed)
453  output.clear();
454  if (input.empty()) return; // ensure we have some values
455  output.reserve(input.size());
456  // get data
457  output.push_back(input[0]);
458  for (size_t i = 1; i < input.size(); ++i) {
459  output.push_back(output[i - 1] + input[i]);
460  }
461 }

469 // recursive indexing decode
470 template<typename SmallInt, typename Int>
471 void BinaryDecoder::recursiveIndexDecode_(const std::vector<SmallInt>& input,
472  std::vector<Int>& output) {
473  // get limits
474  const SmallInt min_int = std::numeric_limits<SmallInt>::min();
475  const SmallInt max_int = std::numeric_limits<SmallInt>::max();
476  // find out size of resulting vector (for speed)
477  size_t out_size = 0;
478  for (size_t i = 0; i < input.size(); ++i) {
479  if (input[i] != min_int && input[i] != max_int) ++out_size;
480  }
481  // reserve space (for speed)
482  output.clear();
483  output.reserve(out_size);
484  // get data
485  Int cur_val = 0;
486  for (size_t i = 0; i < input.size(); ++i) {
487  cur_val += input[i];
488  if (input[i] != min_int && input[i] != max_int) {
489  output.push_back(cur_val);
490  cur_val = 0;
491  }
492  }
493 }
